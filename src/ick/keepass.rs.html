1use crate::credentials::Credential;
2use anyhow::anyhow;
3use keepass::{Database, DatabaseKey, db::Entry};
4use log::trace;
5use std::fs::File;
6
7const USERNAME_KEY: &str = "UserName";
8const PASSWORD_KEY: &str = "Password";
9
10/// Get the name of the keepass group containing the relevant set of credentials.
11fn get_keepass_group_name(admin: bool) -> &'static str {
12    if admin { "Admin" } else { "User" }
13}
14
15fn node_to_credential(node: &Entry, machine: String) -> anyhow::Result<Credential> {
16    Ok(Credential::new(
17        machine,
18        node.get(USERNAME_KEY)
19            .ok_or(anyhow!("Username field not found"))?
20            .to_owned(),
21        node.get(PASSWORD_KEY)
22            .ok_or(anyhow!("Password field not found"))?
23            .to_owned(),
24    ))
25}

27/// Get credentials for the specified machines from keepass.
28pub fn get_credentials_keepass<T: AsRef<str>>(
29    source: &str,
30    key: &str,
31    machines: &[T],
32    admin: bool,
33) -> anyhow::Result<Vec<Credential>> {
34    trace!("Getting credentials from keepass DB at {source}");
35
36    let mut file = File::open(source)?;
37    let key = DatabaseKey::new().with_password(key);
38    let db = Database::open(&mut file, key)
39        .map_err(|e| anyhow!("Failed to open database at {source}: {e}"))?;
40
41    trace!("Keepass database at {source} successfully opened");
42
43    let group_name = get_keepass_group_name(admin);
44
45    trace!("Looking for credentials in group {group_name}");
46
47    machines
48        .iter()
49        .map(AsRef::as_ref)
50        .map(|machine| {
51            let root = db.root();
52            let group = root
53                .group_by_name(group_name)
54                .ok_or(anyhow!("unable to find group {group_name} in {source}",))?;
55            let entry = group.entry_by_name(machine).ok_or(anyhow!(
56                "unable to find credential {group_name}/{machine} in {source}",
57            ))?;
58            node_to_credential(&entry, machine.to_owned())
59                .map_err(|e| anyhow!("Failed to read credential at {group_name}/{machine}: {e}"))
60        })
61        .collect()
62}
