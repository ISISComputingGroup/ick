1//! **i**nstrument-**c**redentials-from-**k**eeper/**k**eepass.
2//!
3//! ick provides a thin wrapper over the windows credential store, injecting appropriate credentials
4//! which are acquired from either keeper or keepass.
5//!
6//! # Installation
7//!
8//! Put the binary `ick.exe` in a directory on your `PATH`.
9//!
10//! `ick` requires environment variables in order to acquire credentials:
11//! - `ICK_CRED_STORE`, should be set to either `keeper` or `keepass`, describing the
12//!   password-manager backend which `ick` will acquire passwords from. This environment
13//!   variable can be set permanently.
14//! - `ICK_KEEPASS_FILE` (keepass only), a path to a keepass (.kdbx) file containing the passwords.
15//!   This environment variable can be set permanently.
16//! - `ICK_KEEPASS_KEY` (keepass only), the decryption password for the keepass database. **This should
17//!   not be set as a permanent environment variable**; it should be manually entered as an env variable
18//!   into a specific shell, and should be cleared (or the shell closed) once administration tasks are complete.
19//! - `ICK_KEEPER_TOKEN` (keeper only), a token used to access the keeper API. **This should
20//!   not be set as a permanent environment variable**; it should be manually entered as an env variable
21//!   into a specific shell, and should be cleared (or the shell closed) once administration tasks are complete.
22//!   This is not supported yet.
23//!
24//! # Usage
25//!
26//! For a description of available `ick` commands and flags, use:
27//! ```
28//! ick help
29//! ```
30//!
31//! For help on a specific subcommand, for example `ick add-creds`, use:
32//! ```
33//! ick help add-creds
34//! ```
35//!
36//! The default Logging level is INFO. This can be increased or decreased one level at a time with the
37//! `-v`/`--verbose` or `-q`/`--quiet` flags, which can be specified multiple times. Logs are printed to stderr.
38//!
39//! # Examples
40//!
41//! Add user-level credentials for `INST1` and `INST2` to the windows credential store, as an unprivileged user or admin user:
42//! ```
43//! ick add-creds -i NDXINST1,NDXINST2
44//! ick add-creds -i NDXINST1,NDXINST2 --admin
45//! ```
46//!
47//! Specify machines in a file instead of on the command line:
48//! ```
49//! ick add-creds -I machines.txt
50//! ```
51//!
52//! Remove credentials for `INST1` and `INST2` from the windows credential store:
53//! ```
54//! ick remove-creds -i NDXINST1,NDXINST2
55//! ```
56//!
57//! Retrieve admin credentials for the specified machine in pretty-JSON format, for use by other tools or scripts:
58//! ```
59//! ick json -I machines.txt --admin --pretty
60//! ```
61//!
62//! With non-pretty output, an example output from `ick json` is:
63//! ```
64//! [{"host":"NDXSOMETHING","username":"NDXSOMETHING\\username","password":"the-password"}]
65//! ```
66//!
67//! # Local development
68//!
69//! You need cargo installed; see <https://rustup.rs/> for first-time install or run `rustup update` to update.
70//!
71//! To run formatter, use `cargo fmt`
72//!
73//! To run linter, use `cargo clippy`
74//!
75//! To run tests, use `cargo test`
76//!
77//! To build in debug configuration, use `cargo build` (the executable will be in `.\target\debug\ick`)
78//!
79//! To build in release configuration, use `cargo build --release` (the executable will be in `.\target\release\ick`)
80//!
81//! To build docs page, use `cargo doc --no-deps` (the docs will be in `.\target\doc\ick`)
82//!
83//! To run the command, use the ick executable which has been built in `target/release` (or `target/debug` for a debug build)

94#[derive(Debug, Args)]
95struct GlobalOpts {
96    #[clap(
97        long = "instruments",
98        short = 'i',
99        value_delimiter = ',',
100        global = true,
101        help = "A comma separated list of machines to apply this command to.
102Conflicts with --instruments-file"
103    )]
104    instruments: Vec<String>,
105
106    #[clap(
107        long = "instruments-file",
108        short = 'I',
109        global = true,
110        help = "Path to a line-delimited text file containing instrument names.
111Lines beginning with '#' are ignored as comments.
112Conflicts with --instruments",
113        conflicts_with = "instruments"
114    )]
115    instruments_file: Option<String>,
116
117    #[clap(
118        long = "admin",
119        short = 'a',
120        global = true,
121        action,
122        help = "Use admin credentials when adding credentials. 
123Defaults to false (unprivileged user), specify this flag to use privileged credentials."
124    )]
125    admin: bool,
126}

141#[derive(Subcommand, Debug)]
142enum Commands {
143    /// Add credentials to the windows credential store
144    #[cfg(target_os = "windows")]
145    AddCreds {},
146
147    /// Remove credentials from the windows credential store
148    #[cfg(target_os = "windows")]
149    RemoveCreds {},

152    /// by external tools or scripts.
153    Json {
154        #[arg(
155            short = 'p',

158            help = "Pretty-print JSON output"

164fn print_json<T: AsRef<str>>(instruments: &[T], admin: bool, pretty: bool) -> anyhow::Result<()> {
165    let creds = credentials::get_credentials(instruments, admin, None)?;
166    trace!("Serialising credentials to JSON");
167
168    let json = if pretty {
169        serde_json::to_string_pretty(&creds)
170    } else {
171        serde_json::to_string(&creds)
172    }?;
173    println!("{json}");
174    Ok(())
175}

177fn load_instruments_from_file(file_contents: &str) -> Vec<String> {
178    file_contents
179        .lines()
180        .map(|line| line.trim())
181        .filter(|line| !line.is_empty() && !line.starts_with("#"))
182        .map(str::to_owned)
183        .collect()
184}
185
186fn get_machines(args: &App) -> anyhow::Result<Vec<String>> {
187    let machines: Vec<String> = if let Some(filename) = &args.global_opts.instruments_file {
188        let file_contents = std::fs::read_to_string(filename)
189            .with_context(|| format!("Instrument list file at '{filename}' could not be read"))?;
190        load_instruments_from_file(&file_contents)
191    } else {
192        args.global_opts.instruments.clone()
193    };
194
195    if machines.is_empty() {
196        bail!("No machines specified")
197    }
198    debug!("Instruments: {:?}", machines);
199    Ok(machines)
200}
201
202#[doc(hidden)]
203pub fn run() -> anyhow::Result<()> {
204    let args = App::parse();
205
206    // Setup logging
207    env_logger::Builder::new()
208        .filter_level(args.verbosity.into())
209        .init();
210
211    trace!("Logging started");
212
213    let machines = get_machines(&args)?;
214
215    match args.command {
216        #[cfg(target_os = "windows")]
217        Commands::AddCreds {} => wincred::add_win_creds(&machines, args.global_opts.admin),
218
219        #[cfg(target_os = "windows")]
220        Commands::RemoveCreds {} => wincred::remove_win_creds(&machines),
221
222        Commands::Json { pretty } => print_json(&machines, args.global_opts.admin, pretty),
223    }
224}

231    #[test]
232    fn test_load_instruments_from_file() {
233        let file_contents = "
234foo
235bar
236# comment
237    baz
238    # another_comment
239
240# another comment after a blank line
241";
242        assert_eq!(
243            load_instruments_from_file(&file_contents),
244            vec!["foo", "bar", "baz"]
245        )
246    }
247
248    #[test]
249    fn test_get_machines_with_no_provided_instruments() {

251            verbosity: Verbosity::default(),
252            global_opts: GlobalOpts {
253                instruments: vec![],
254                instruments_file: None,
255                admin: false,
256            },
257            command: Commands::Json { pretty: false },

259
260        assert!(result.is_err_and(|e| { e.to_string().contains("No machines specified") }));
