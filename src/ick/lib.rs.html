1//! **i**nstrument-**c**redentials-from-**k**eeper/**k**eepass.
2//!
3//! ick provides a thin wrapper over the windows credential store, injecting appropriate credentials
4//! which are acquired from either keeper or keepass.
5//!
6//! # Installation
7//!
8//! Put the binary `ick.exe` in a directory on your `PATH`.
9//!
10//! `ick` requires environment variables in order to acquire credentials:
11//! - `ICK_CRED_STORE`, should be set to either `keeper` or `keepass`, describing the
12//!   password-manager backend which `ick` will acquire passwords from. This environment
13//!   variable can be set permanently.
14//! - `ICK_KEEPASS_FILE` (keepass only), a path to a keepass (.kdbx) file containing the passwords.
15//!   This environment variable can be set permanently.
16//! - `ICK_KEEPASS_KEY` (keepass only), the decryption password for the keepass database. **This should
17//!   not be set as a permanent environment variable**; it should be manually entered as an env variable
18//!   into a specific shell, and should be cleared (or the shell closed) once administration tasks are complete.
19//! - `ICK_KEEPER_TOKEN` (keeper only), a token used to access the keeper API. **This should
20//!   not be set as a permanent environment variable**; it should be manually entered as an env variable
21//!   into a specific shell, and should be cleared (or the shell closed) once administration tasks are complete.
22//!   To generate a token, log into keeper as an account with access to keeper secrets manager, go to the
23//!   'secrets manager' tab, select the 'ick' application, and click 'add device'. Set the 'device name' to
24//!   your name, select 'configuration file (base64)' as the method. This will generate a base64 string which
25//!   you can use as the `ICK_KEEPER_TOKEN`. Store this string in your personal password manager.
26//!
27//! # Usage
28//!
29//! For a description of available `ick` commands and flags, use:
30//! ```
31//! ick help
32//! ```
33//!
34//! For help on a specific subcommand, for example `ick add-creds`, use:
35//! ```
36//! ick help add-creds
37//! ```
38//!
39//! The default Logging level is WARN. This can be increased or decreased one level at a time with the
40//! `-v`/`--verbose` or `-q`/`--quiet` flags, which can be specified multiple times. Logs are printed to stderr.
41//!
42//! # Examples
43//!
44//! Add user-level credentials for `INST1` and `INST2` to the windows credential store, as an unprivileged user or admin user:
45//! ```
46//! ick add-creds -i NDXINST1,NDXINST2
47//! ick add-creds -i NDXINST1,NDXINST2 --admin
48//! ```
49//!
50//! Specify machines in a file instead of on the command line:
51//! ```
52//! ick add-creds -I machines.txt
53//! ```
54//!
55//! Remove credentials for `INST1` and `INST2` from the windows credential store:
56//! ```
57//! ick remove-creds -i NDXINST1,NDXINST2
58//! ```
59//!
60//! Retrieve admin credentials for the specified machine in pretty-JSON format, for use by other tools or scripts:
61//! ```
62//! ick json -I machines.txt --admin --pretty
63//! ```
64//!
65//! With non-pretty output, an example output from `ick json` is:
66//! ```
67//! [{"host":"NDXSOMETHING","username":"NDXSOMETHING\\username","password":"the-password"}]
68//! ```
69//!
70//! # Local development
71//!
72//! You need cargo installed; see <https://rustup.rs/> for first-time install or run `rustup update` to update.
73//!
74//! To run formatter, use `cargo fmt`
75//!
76//! To run linter, use `cargo clippy`
77//!
78//! To run tests, use `cargo test`
79//!
80//! To build in debug configuration, use `cargo build` (the executable will be in `.\target\debug\ick`)
81//!
82//! To build in release configuration, use `cargo build --release` (the executable will be in `.\target\release\ick`)
83//!
84//! To build docs page, use `cargo doc --no-deps` (the docs will be in `.\target\doc\ick`)
85//!
86//! To run the command, use the ick executable which has been built in `target/release` (or `target/debug` for a debug build)

99#[derive(Debug, Args)]
100struct InstrumentOpts {
101    #[clap(
102        long = "instruments",
103        short = 'i',
104        value_delimiter = ',',
105        global = true,
106        help = "A comma separated list of machines to apply this command to.
107Conflicts with --instruments-file"
108    )]
109    instruments: Vec<String>,
110
111    #[clap(
112        long = "instruments-file",
113        short = 'I',
114        global = true,
115        help = "Path to a line-delimited text file containing instrument names.
116Lines beginning with '#' are ignored as comments.
117Conflicts with --instruments",
118        conflicts_with = "instruments"
119    )]
120    instruments_file: Option<String>,
121
122    #[clap(
123        long = "admin",
124        short = 'a',
125        global = true,
126        action,
127        help = "Use admin credentials when adding credentials. 
128Defaults to false (unprivileged user), specify this flag to use privileged credentials."
129    )]
130    admin: bool,
131}

143#[derive(Subcommand, Debug)]
144enum Commands {
145    /// Add credentials to the windows credential store
146    #[cfg(target_os = "windows")]
147    AddCreds {
148        #[clap(flatten)]
149        instrument_opts: InstrumentOpts,
150    },
151
152    /// Remove credentials from the windows credential store
153    #[cfg(target_os = "windows")]
154    RemoveCreds {
155        #[clap(flatten)]
156        instrument_opts: InstrumentOpts,
157    },
158
159    /// Retrieve credentials for all specified machines in JSON format, for use
160    /// by external tools or scripts.
161    Json {
162        #[clap(flatten)]
163        instrument_opts: InstrumentOpts,
164
165        #[arg(
166            short = 'p',
167            long = "pretty",
168            action,
169            help = "Pretty-print JSON output"
170        )]
171        pretty: bool,
172    },
173}

175fn print_json<T: AsRef<str>>(instruments: &[T], admin: bool, pretty: bool) -> anyhow::Result<()> {
176    let creds = credentials::get_credentials(instruments, admin, None)?;
177    trace!("Serialising credentials to JSON");
178
179    let json = if pretty {
180        serde_json::to_string_pretty(&creds)
181    } else {
182        serde_json::to_string(&creds)
183    }?;
184    println!("{json}");
185    Ok(())
186}

188fn load_instruments_from_file(file_contents: &str) -> Vec<String> {
189    file_contents
190        .lines()
191        .map(|line| line.trim())
192        .filter(|line| !line.is_empty() && !line.starts_with("#"))
193        .map(str::to_owned)
194        .collect()
195}
196
197fn get_machines(args: &InstrumentOpts) -> anyhow::Result<Vec<String>> {
198    let machines: Vec<String> = if let Some(filename) = &args.instruments_file {
199        let file_contents = std::fs::read_to_string(filename)
200            .with_context(|| format!("Instrument list file at '{filename}' could not be read"))?;
201        load_instruments_from_file(&file_contents)
202    } else {
203        args.instruments.clone()
204    };
205
206    if machines.is_empty() {
207        bail!("No machines specified")
208    }
209    debug!("Instruments: {:?}", machines);
210    Ok(machines)
211}

213#[doc(hidden)]
214pub fn run() -> anyhow::Result<()> {
215    let args = App::parse();
216
217    // Setup logging
218    env_logger::Builder::new()
219        .filter_level(args.verbosity.into())
220        .init();
221
222    trace!("Logging started");
223
224    match args.command {
225        #[cfg(target_os = "windows")]
226        Commands::AddCreds { instrument_opts } => {
227            let machines = get_machines(&instrument_opts)?;
228            wincred::add_win_creds(&machines, instrument_opts.admin)
229        }
230
231        #[cfg(target_os = "windows")]
232        Commands::RemoveCreds { instrument_opts } => {
233            let machines = get_machines(&instrument_opts)?;
234            wincred::remove_win_creds(&machines)
235        }
236
237        Commands::Json {
238            instrument_opts,
239            pretty,
240        } => {
241            let machines = get_machines(&instrument_opts)?;
242            print_json(&machines, instrument_opts.admin, pretty)
243        }
244    }
245}

251    #[test]
252    fn test_load_instruments_from_file() {
253        let file_contents = "
254foo
255bar
256# comment
257    baz
258    # another_comment
259
260# another comment after a blank line
261";
262        assert_eq!(
263            load_instruments_from_file(&file_contents),

266    }

273            admin: false,
274        });
