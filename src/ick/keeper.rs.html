13fn record_to_credential(record: &Record) -> anyhow::Result<Credential> {
14    let host = record.title.clone();
15    let username = record
16        .get_standard_field_value(StandardFieldTypeEnum::LOGIN.get_type(), true)?
17        .as_str()
18        .ok_or(anyhow!("failed to read username for {host}"))?
19        .to_owned();
20    let password = record
21        .get_standard_field_value(StandardFieldTypeEnum::PASSWORD.get_type(), true)?
22        .as_str()
23        .ok_or(anyhow!("failed to read password for {host}"))?
24        .to_owned();
25
26    Ok(Credential::new(host, username, password))
27}

35fn get_secrets_manager(keeper_key: &str) -> anyhow::Result<SecretsManager> {
36    let decoded_key = BASE64_STANDARD.decode(keeper_key)?;
37    let config_contents = String::from_utf8(decoded_key)?;
38    let config = InMemoryKeyValueStorage::new_config_storage(Some(config_contents))?;
39    let client_options = ClientOptions::new_client_options(config);
40    SecretsManager::new(client_options).map_err(|e| anyhow!("{e}"))
41}

43fn get_allowed_folder_uids(folders: &[KeeperFolder], admin: bool) -> HashSet<String> {
44    folders
45        .iter()
46        .filter(|f| (!admin) ^ is_admin_folder(f))
47        .map(|f| f.folder_uid.clone())
48        .collect()
49}

51pub fn get_credentials_keeper<T: AsRef<str>>(
52    keeper_key: &str, // Really a base64'd version of a config file.
53    machines: &[T],
54    admin: bool,
55) -> anyhow::Result<Vec<Credential>> {
56    let mut secrets_manager = get_secrets_manager(keeper_key)?;
57
58    let all_secrets = secrets_manager.get_secrets(vec![])?;
59
60    let allowed_folder_uids = get_allowed_folder_uids(&secrets_manager.get_folders()?, admin);
61
62    let all_creds = all_secrets
63        .iter()
64        .filter(|r| allowed_folder_uids.contains(&r.folder_uid))
65        .map(record_to_credential)
66        .map(|cred| cred.map(|c| (c.host().to_owned(), c)))
67        .collect::<anyhow::Result<HashMap<_, _>>>()?;
68
69    machines
70        .iter()
71        .map(AsRef::as_ref)
72        .map(|machine| {
73            all_creds
74                .get(machine)
75                .cloned()
76                .ok_or(anyhow!("Failed to get credential for {machine}"))
77        })
78        .collect()
79}
